# Pseudocode Luas dan Keliling Lingkaran #

Deklarasi double pi, jari-jari, luas, keliling\

print "nilai jari-jari"\

luas = pi * jari-jari * jari jari\
keliling = 2 * pi * jari-jari

print "luas" dan "keliling"
