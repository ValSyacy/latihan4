# Pseudocode Luas dan Keliling Lingkaran #

Deklarasi double π, jari-jari, luas, keliling\
input nilai π = 3.14

print "nilai jari-jari"\
input nilai jari-jari

luas = π * jari-jari * jari-jari\
keliling = 2 * π * jari-jari

print "luas" dan "keliling"
